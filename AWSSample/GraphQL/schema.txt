type Diary {
	id: ID!
	title: String
	author: String
}

type Mutation {
	insertDiary(id: ID!, title: String, author: String): Diary
}

type Query {
	diaries(author: String): [Diary]
}

type Subscription {
	onSubscribe(author: String): Diary
		@aws_subscribe(mutations: ["insertDiary"])
}

schema {
	query: Query
	mutation: Mutation
	subscription: Subscription
}
